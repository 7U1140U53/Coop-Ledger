// 2. CONTEXTUAL ROUTER ENGINE (Determines View Permissions)
function initializeAppRouting() {
    const email = state.currentUser.email;
    const isTreasurer = email.endsWith('@coreledger.com') || email.endsWith('@coop.com');

    const userContextText = document.getElementById('user-context-text');
    const roleStatusDot = document.getElementById('role-status-dot');

    if (isTreasurer) {
        // Set context indicator bar metadata
        userContextText.innerHTML = `Logged in as <strong>${state.currentUser.fullName}</strong>. <button id="view-toggle-link" class="text-indigo-400 hover:underline font-semibold ml-2">Switch to My Member Deposit Form →</button>`;
        roleStatusDot.className = "w-2 h-2 rounded-full bg-emerald-400 animate-pulse";

        // Expose components
        document.getElementById('prd-treasurer-interface').classList.remove('hidden');
        document.getElementById('prd-member-interface').classList.add('hidden');

        // Attach interactive quick-switch listener
        document.getElementById('view-toggle-link').addEventListener('click', toggleTreasurerHats);
    } else {
        userContextText.innerHTML = `Logged in as Member: <strong>${state.currentUser.fullName}</strong> (${state.currentUser.membershipId})`;
        roleStatusDot.className = "w-2 h-2 rounded-full bg-indigo-400";

        document.getElementById('prd-treasurer-interface').classList.add('hidden');
        document.getElementById('prd-member-interface').classList.remove('hidden');
    }

    renderDashboardMetrics();
    renderClearingFeedTable();
}

// Interactive utility letting the Treasurer jump frames to pay their own contributions
function toggleTreasurerHats() {
    const memberInterface = document.getElementById('prd-member-interface');
    const toggleLink = document.getElementById('view-toggle-link');

    if (memberInterface.classList.contains('hidden')) {
        memberInterface.classList.remove('hidden');
        toggleLink.textContent = "← Return to Admin Clearing Feed";
    } else {
        memberInterface.classList.add('hidden');
        toggleLink.textContent = "Switch to My Member Deposit Form →";
    }
}

// 3. STATISTICAL COMPONENT RENDERING
function renderDashboardMetrics() {
    // Formatting numbers to crisp Nigerian Naira displays natively
    const nairaFormat = new Intl.NumberFormat('en-NG', { style: 'currency', currency: 'NGN' });

    document.getElementById('metric-approved').textContent = nairaFormat.format(state.approvedPool);
    document.getElementById('metric-pending').textContent = nairaFormat.format(state.pipelinePending);
    document.getElementById('metric-flagged').textContent = nairaFormat.format(state.flaggedExceptions);
}

// 4. TRANSACTION RECONCILIATION QUEUE DATA-MAPPER

    const tbody = document.getElementById('clearing-feed-rows');
    const queueCountBadge = document.getElementById('queue-count');

    const pendingTransactions = state.contributions.filter(tx => tx.status === 'PENDING');
    queueCountBadge.textContent = `${pendingTransactions.length} entries remaining`;

    tbody.innerHTML = '';

    if (pendingTransactions.length === 0) {
        tbody.innerHTML = `
            <tr>
                <td colspan="6" class="p-8 text-center text-xs text-slate-500 font-medium tracking-wide">
                    🎉 All pipeline inflows perfectly reconciled against bank log files!
                </td>
            </tr>`;

    }

    const nairaFormat = new Intl.NumberFormat('en-NG', { style: 'currency', currency: 'NGN' });

    pendingTransactions.forEach(tx => {

        row.className = "text-xs hover:bg-slate-800/20 transition duration-150";

                <p class="text-[10px] text-slate-500 font-mono mt-0.5">${tx.membershipId}</p>
            </td>
            <td class="p-4 text-slate-300 font-medium">${tx.cycleTarget}</td>
            <td class="p-4 text-right font-bold text-slate-100">${nairaFormat.format(tx.amount)}</td>
            <td class="p-4 font-mono text-[11px] space-y-0.5">

                <p class="text-slate-500">${tx.reference}</p>
            </td>
            <td class="p-4 text-center">
                <span class="inline-flex items-center space-x-1 px-2 py-1 bg-slate-800 border border-slate-700 rounded text-[10px] font-mono text-slate-400 cursor-zoom-in hover:border-slate-500 transition">

// 5. SECURE ACTION RESOLUTION EVENT HANDLERS
document.addEventListener('click', function (event) {
    const target = event.target;
    const txId = target.getAttribute('data-id');
    const action = target.getAttribute('data-action');

    if (!txId) return;

    const txIndex = state.contributions.findIndex(t => t.id === txId);
    if (txIndex === -1) return;
    const targetTx = state.contributions[txIndex];

    if (action === 'approve') {
        // Run verification accounting pipeline
        state.approvedPool += targetTx.amount;
        state.pipelinePending -= targetTx.amount;
        targetTx.status = 'APPROVED';

        renderDashboardMetrics();
        renderClearingFeedTable();
    }
    else if (action === 'flag') {
        // Open the modal capture element
        activeTargetTxId = txId;
        document.getElementById('flag-reason-input').value = '';
        document.getElementById('flag-modal').classList.remove('hidden');
    }
});

// Modal Confirmation Management
document.getElementById('modal-confirm-btn').addEventListener('click', function () {
    if (!activeTargetTxId) return;

    const txIndex = state.contributions.findIndex(t => t.id === activeTargetTxId);
    if (txIndex !== -1) {
        const targetTx = state.contributions[txIndex];
        const reason = document.getElementById('flag-reason-input').value.trim();

        // Run exception tracking mechanics
        state.flaggedExceptions += targetTx.amount;
        state.pipelinePending -= targetTx.amount;
        targetTx.status = 'FLAGGED';
        targetTx.exceptionNotes = reason || "Unspecified validation fault.";

        // Hide modal & execute table re-render
        document.getElementById('flag-modal').classList.add('hidden');
        activeTargetTxId = null;

        renderDashboardMetrics();
        renderClearingFeedTable();
    }
});

document.getElementById('modal-cancel-btn').addEventListener('click', function () {
    document.getElementById('flag-modal').classList.add('hidden');
    activeTargetTxId = null;
});
