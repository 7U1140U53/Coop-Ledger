// Helper function to fetch user profile metadata safely from Supabase without breaking on missing tables
async function fetchProfileAndSynchronize(user) {
    try {
        const { data: profile, error } = await supabaseClient
            .from('profiles')
            .select('full_name, membership_id, role')
            .eq('id', user.id)
            .single();

        if (error && error.code !== 'PGRST116') throw error;

        state.currentUser = {
            email: user.email,
            fullName: profile?.full_name || user.email.split('@')[0],
            membershipId: profile?.membership_id || "COOP-2026-UNKNOWN",
            role: profile?.role || 'MEMBER'
        };
    } catch (err) {
        console.error("Profile sync exception:", err);
        state.currentUser = {
            email: user.email,
            fullName: user.email.split('@')[0],
            membershipId: "COOP-2026-UNKNOWN",
            role: 'MEMBER'
        };
    }
}

// Helper panel layout switcher to toggle tabs cleanly without complex full page router refreshes
function switchDashboardPanel(panel) {
    const linkPending = document.getElementById('menu-link-pending');
    const linkHistory = document.getElementById('menu-link-history');
    const panelPending = document.getElementById('panel-pending-feed');
    const panelHistory = document.getElementById('panel-reconciliation-history');

    if (!linkPending || !linkHistory || !panelPending || !panelHistory) return;

    if (panel === 'pending') {
        linkPending.className = "w-full flex items-center space-x-3 px-3 py-2.5 bg-slate-900 border border-slate-800 text-slate-100 rounded-xl text-xs font-semibold tracking-wide transition";
        linkHistory.className = "w-full flex items-center space-x-3 px-3 py-2.5 text-slate-400 hover:bg-slate-900/50 hover:text-slate-200 rounded-xl text-xs font-semibold tracking-wide transition";
        panelPending.classList.remove('hidden');
        panelHistory.classList.add('hidden');
    } else {
        linkHistory.className = "w-full flex items-center space-x-3 px-3 py-2.5 bg-slate-900 border border-slate-800 text-slate-100 rounded-xl text-xs font-semibold tracking-wide transition";
        linkPending.className = "w-full flex items-center space-x-3 px-3 py-2.5 text-slate-400 hover:bg-slate-900/50 hover:text-slate-200 rounded-xl text-xs font-semibold tracking-wide transition";
        panelPending.classList.add('hidden');
        panelHistory.classList.remove('hidden');
        renderReconciliationHistoryTable();
    }
}

// 4. PENDING ENTRIES CONTROLLER
function renderClearingFeedTable() {
    const tbody = document.getElementById('clearing-feed-rows');
    const queueCountBadge = document.getElementById('queue-count');

    if (!tbody) return;

    const pendingTransactions = state.contributions.filter(tx => tx.status === 'PENDING');
    if (queueCountBadge) {
        queueCountBadge.textContent = `${pendingTransactions.length} entries remaining`;
    }

    tbody.innerHTML = '';

    if (pendingTransactions.length === 0) {
        tbody.innerHTML = `
            <tr>
                <td colspan="6" class="p-8 text-center text-xs text-slate-500 font-medium tracking-wide">
                    🎉 All pipeline inflows perfectly reconciled!
                </td>
            </tr>`;
        return;
    }

    const nairaFormat = new Intl.NumberFormat('en-NG', { style: 'currency', currency: 'NGN' });

    pendingTransactions.forEach(tx => {
        const row = document.createElement('tr');
        row.className = "text-xs hover:bg-slate-800/20 transition duration-150";
        row.id = `row-${tx.id}`;

        row.innerHTML = `
            <td class="p-4">
                <p class="font-bold text-slate-200">${tx.memberName}</p>
                <p class="text-[10px] text-slate-500 font-mono mt-0.5">${tx.membershipId}</p>
            </td>
            <td class="p-4 text-slate-300 font-medium">${tx.cycleTarget}</td>
            <td class="p-4 text-right font-bold text-slate-100">${nairaFormat.format(tx.amount)}</td>
            <td class="p-4 font-mono text-[11px] space-y-0.5">
                <p class="text-slate-300 font-semibold">${tx.bankName} (${tx.accountName})</p>
                <p class="text-slate-500">${tx.reference}</p>
            </td>
            <td class="p-4 text-center">
                <span class="inline-flex items-center space-x-1 px-2 py-1 bg-slate-800 border border-slate-700 rounded text-[10px] font-mono text-slate-400">
                    <span>${tx.receiptFile}</span>
                </span>
            </td>
            <td class="p-4 text-right space-x-2">
                <button data-id="${tx.id}" data-action="flag" class="px-3 py-1.5 bg-rose-950/40 hover:bg-rose-900/60 border border-rose-800/50 hover:border-rose-700 text-rose-300 font-bold rounded-lg transition text-[11px]">
                    Flag Fault

                </button>
            </td>

    });
}

// 5. RECONCILED ARCHIVE ENTRY HISTORY COMPONENT
function renderReconciliationHistoryTable() {
    const tbody = document.getElementById('history-feed-rows');
    if (!tbody) return;

    const archivedTransactions = state.contributions.filter(tx => tx.status === 'APPROVED' || tx.status === 'FLAGGED');
    tbody.innerHTML = '';

    if (archivedTransactions.length === 0) {
        tbody.innerHTML = `<tr><td colspan="5" class="p-8 text-center text-xs text-slate-500 font-medium font-sans">No archived reconciliation logs recorded yet.</td></tr>`;
        return;
    }

    const nairaFormat = new Intl.NumberFormat('en-NG', { style: 'currency', currency: 'NGN' });

    archivedTransactions.forEach(tx => {
        const row = document.createElement('tr');
        row.className = "text-xs hover:bg-slate-800/10 transition";

        let badgeMarkup = "";
        if (tx.status === 'APPROVED') {
            badgeMarkup = `<span class="inline-block px-2.5 py-1 bg-emerald-950/40 border border-emerald-800/50 text-emerald-400 text-[10px] rounded-lg font-mono font-bold tracking-wider uppercase">✓ Reconciled</span>`;
        } else {
            badgeMarkup = `
                <div class="text-right space-y-1">
                    <span class="inline-block px-2.5 py-1 bg-rose-950/40 border border-rose-800/50 text-rose-400 text-[10px] rounded-lg font-mono font-bold tracking-wider uppercase">⚠ Flagged Fault</span>
                    <p class="text-[10px] text-rose-400 font-medium italic">${tx.exceptionNotes || 'Validation discrepancy.'}</p>
                </div>`;
        }

        row.innerHTML = `
            <td class="p-4">
                <p class="font-bold text-slate-300">${tx.memberName}</p>
                <p class="text-[10px] text-slate-500 font-mono mt-0.5">${tx.membershipId}</p>
            </td>
            <td class="p-4 text-slate-400">${tx.cycleTarget}</td>
            <td class="p-4 text-right font-bold text-slate-100">${nairaFormat.format(tx.amount)}</td>
            <td class="p-4 font-mono text-[11px] text-slate-500 space-y-0.5">
                <p class="text-slate-400 font-medium">${tx.bankName} (${tx.accountName})</p>
                <p>REF: ${tx.reference}</p>
            </td>
            <td class="p-4 text-right">
                ${badgeMarkup}
            </td>
        `;
        tbody.appendChild(row);
    });
}

// 6. ACTION RESOLUTION HANDLERS
document.addEventListener('click', function (event) {
    const target = event.target;

    // PASSWORD VISIBILITY HOOK
    if (target && target.classList.contains('password-toggle-btn')) {
        const passwordInput = target.previousElementSibling;
        if (passwordInput) {
            if (passwordInput.type === 'password') {
                passwordInput.type = 'text';
                target.textContent = 'Hide';
            } else {
                passwordInput.type = 'password';
                target.textContent = 'Show';
            }
        }
        return;
    }

    // SIDEBAR NAVIGATION PANEL TRIGGERS
    if (target && (target.id === 'menu-link-pending' || target.closest('#menu-link-pending'))) {
        switchDashboardPanel('pending');
        return;
    }
    if (target && (target.id === 'menu-link-history' || target.closest('#menu-link-history'))) {
        switchDashboardPanel('history');
        return;
    }

    const txId = target.getAttribute('data-id');
    const action = target.getAttribute('data-action');
    if (!txId) return;

    const txIndex = state.contributions.findIndex(t => t.id === txId);
    if (txIndex === -1) return;
    const targetTx = state.contributions[txIndex];

    if (action === 'approve') {
        state.approvedPool += targetTx.amount;
        state.pipelinePending -= targetTx.amount;
        targetTx.status = 'APPROVED';

        renderDashboardMetrics();
        renderClearingFeedTable();
    }
    else if (action === 'flag') {
        activeTargetTxId = txId;
        document.getElementById('flag-reason-input').value = '';
        document.getElementById('flag-modal').classList.remove('hidden');
    }
});

// Modal Actions Configuration Hooks
document.getElementById('modal-confirm-btn').addEventListener('click', function () {
    if (!activeTargetTxId) return;

    const txIndex = state.contributions.findIndex(t => t.id === activeTargetTxId);
    if (txIndex !== -1) {
        const targetTx = state.contributions[txIndex];
        const reason = document.getElementById('flag-reason-input').value.trim();

        state.flaggedExceptions += targetTx.amount;
        state.pipelinePending -= targetTx.amount;
        targetTx.status = 'FLAGGED';
        targetTx.exceptionNotes = reason || "Unspecified validation fault.";

        document.getElementById('flag-modal').classList.add('hidden');
        activeTargetTxId = null;

        renderDashboardMetrics();
        renderClearingFeedTable();
    }
});

document.getElementById('modal-cancel-btn').addEventListener('click', function () {
    document.getElementById('flag-modal').classList.add('hidden');
    activeTargetTxId = null;
});

// Global Core Wire Hook Initialization
document.addEventListener('DOMContentLoaded', async function () {
    try {
        const { data: { session } } = await supabaseClient.auth.getSession();

        if (session && session.user) {
            await fetchProfileAndSynchronize(session.user);
            navigateTo('screen-dashboard');
            initializeAppRouting();
        } else {
            navigateTo('screen-landing');
        }
    } catch (error) {
        console.error('Session boot failed:', error);
        navigateTo('screen-landing');
    }

    // Static Base Controls Event Hooks 
    document.getElementById('landing-login-btn').addEventListener('click', function () { navigateTo('screen-auth'); });
    document.getElementById('landing-register-btn').addEventListener('click', function () { navigateTo('screen-register'); });
    document.getElementById('auth-back-btn').addEventListener('click', function () { navigateTo('screen-landing'); });
    document.getElementById('register-back-btn').addEventListener('click', function () { navigateTo('screen-landing'); });

    // Secure Authorization Forms Processing Hook
    document.getElementById('secure-login-form').addEventListener('submit', async function (e) {
        e.preventDefault();

        const emailInput = document.getElementById('auth-email-input');
        const passwordInput = document.getElementById('auth-password-input');
        const submitBtn = document.getElementById('auth-submit-btn');

        emailInput.disabled = true;
        passwordInput.disabled = true;
        submitBtn.disabled = true;
        submitBtn.textContent = "Authenticating Context Base...";

        const email = emailInput.value;
        const password = passwordInput.value;

        try {
            const { data, error } = await supabaseClient.auth.signInWithPassword({ email, password });
            if (error) throw error;

            await fetchProfileAndSynchronize(data.user);

            emailInput.value = '';
            passwordInput.value = '';
            emailInput.disabled = false;
            passwordInput.disabled = false;
            submitBtn.disabled = false;
            submitBtn.textContent = "Initialize Session";

            navigateTo('screen-dashboard');
            initializeAppRouting();

        } catch (error) {
            console.error('Authentication Error Context:', error);

            emailInput.disabled = false;
            passwordInput.disabled = false;
            submitBtn.disabled = false;
            submitBtn.textContent = "Initialize Session";

            const existingError = document.getElementById('auth-error-message');
            if (existingError) existingError.remove();

            const errorDiv = document.createElement('div');
            errorDiv.id = 'auth-error-message';
            errorDiv.className = 'mt-4 p-3 bg-rose-900/20 border border-rose-700/50 rounded-xl text-rose-400 text-xs';
            errorDiv.innerHTML = `<span>Authentication failed: ${error.message}</span>`;
            document.getElementById('secure-login-form').appendChild(errorDiv);
        }
    });

    // Secure Registrations Form Submission Hook
    document.getElementById('secure-registration-form').addEventListener('submit', async function (e) {
        e.preventDefault();

        const nameInput = document.getElementById('reg-name-input');
        const emailInput = document.getElementById('reg-email-input');
        const passwordInput = document.getElementById('reg-password-input');
        const submitBtn = document.getElementById('reg-submit-btn');

        nameInput.disabled = true;
        emailInput.disabled = true;
        passwordInput.disabled = true;
        submitBtn.disabled = true;
        submitBtn.textContent = "Registering Context Entity...";

        const name = nameInput.value;
        const email = emailInput.value;
        const password = passwordInput.value;

        try {
            const { data: authData, error: signupError } = await supabaseClient.auth.signUp({ email, password });
            if (signupError) throw signupError;

            const generatedId = "COOP-2026-" + Math.floor(1000 + Math.random() * 9000);

            const { error: profileError } = await supabaseClient.from('profiles').insert([{
                id: authData.user.id,
                full_name: name,
                membership_id: generatedId,
                role: 'MEMBER'
